#include<iostream>
using namespace std;
int main()
{
    int a,b,c;
    cin>>a>>b>>c;
    if(a==b && b==c && a==c)
    {
        cout<<"Equilateral Triangle";
    }
    else if(a!=b && b!=c && a!=c)
    {
        cout<<"Scalene Triangle";
    }
    else if(a==b && b!=c && a!=c)
    {
        cout<<"Isosceles Triangle";
    }
}
