#include <iostream>
#include <iomanip>
using namespace std;

int main() {
    float w, h, bmi;
    cin >> w >> h;
    w = w * 0.453592; // Convert weight from pounds to kilograms
    h = h * 0.0254;   // Convert height from inches to meters
    bmi = w / (h * h);
    cout << "BMI: " << fixed << setprecision(2) << bmi;
    return 0;
}
